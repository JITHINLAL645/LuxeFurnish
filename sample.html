const cancelOrder = async (req, res) => { try { const { orderId } = req.params; const userId = req.session.userId; if (!userId) { return res.status(401).json({ message: "Unauthorized" }); } const order = await Order.findOne({ orderId, user: userId }); if (!order) { return res.status(404).json({ message: "Order not found" }); } if (order.orderStatus === 'Cancelled') { return res.status(400).json({ message: "Order is already cancelled" }); } if (['Shipped', 'Delivered'].includes(order.orderStatus)) { return res.status(400).json({ message: "Cannot cancel order at this stage" }); } order.orderStatus = 'Cancelled'; order.orderStatusTimestamps.cancelled = new Date(); for (const item of order.items) { await Product.findByIdAndUpdate(item.product, { $inc: { stock: item.quantity } }); } if (order.paymentStatus === "Paid") { const refund = order.totalAmount; let wallet = await Wallet.findOne({ user: userId }); if (!wallet) { wallet = new Wallet({ user: order.user, balanceAmount: 0, wallet_history: [], }); } wallet.balanceAmount += refund; wallet.wallet_history.push({ date: new Date(), amount: refund, description: Refund for cancelled item (Order ID: ${order.orderId}), transactionType: "credited", }); await wallet.save(); } await order.save(); res.status(200).json({ message: "Order cancelled successfully" }); } catch (error) { console.error("Error cancelling order:", error); res.status(500).json({ message: "Internal Server Error" }); } }; userRouter.post('/orders/:orderId',isBlockAuth,userAuthentication,orderController.cancelOrder);
